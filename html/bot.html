Страница Bot-a на Aiogram3 (bot.html) в домашней папке /home/...

Для доступа Nginx к вашей домашней папке /home/...

Необходимо пользователя www-data добавить в группу своего пользователя:

sudo usermod -aG marat www-data

Файл конфигурации /etc/nginx/sites-available/nginx_bot.conf

        
            server {
                listen 4431 ssl;
                listen [::]:4431 ssl;

                ssl_certificate       /etc/ssl/nginx/Bot_IP.crt;
                ssl_certificate_key   /etc/ssl/nginx/Bot_IP.key;

                server_name localhost;
            #    server_name localhost1;

            #    root /var/www/html;

                location /Aiogram3 {
                    root /home/marat/PycharmProjects;
                    index bot.html;

            #        proxy_set_header Host $http_host;
            #        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            #        proxy_redirect off;
            #        proxy_buffering off;
            #        proxy_pass http://127.0.0.1:8080;
                }
            }
