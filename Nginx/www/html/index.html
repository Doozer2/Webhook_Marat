Nginx: Мой текущий сайт

Расположен в корне /var/www/html

Файл конфигурации /etc/nginx/sites-available/sv.conf

        
            server {
                listen 443 ssl;
                listen [::]:443 ssl default_server;

                ssl_certificate       /etc/ssl/nginx/Bot_IP.crt;
                ssl_certificate_key   /etc/ssl/nginx/Bot_IP.key;

                server_name localhost 127.0.0.1;

                root /var/www/html;

                location / {
                }

                location /ind/ {
                    root /var/www;
                    index ind.html;
                }

            }
